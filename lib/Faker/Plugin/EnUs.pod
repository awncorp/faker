This package is meant to be subclassed.

  package Faker::Plugin::EnUs::UserHandle;

  use base 'Faker::Plugin::EnUs';

  sub execute {
    my ($self) = @_;

    return $self->process('@?{{person_last_name}}####');
  }

  package main;

  use Faker;

  my $faker = Faker->new('en-us');

  # bless(..., "Faker")

  my $result = $faker->user_handle;
